## 1. BACKEND API (main.py)
# Built with FastAPI to serve ROI research data to your Dashboard.

from fastapi import FastAPI
from pydantic import BaseModel
from typing import Dict, Any

app = FastAPI()

# Data model matching data_contract.json
class ReportPayload(BaseModel):
    module_name: str
    data_payload: Dict[str, Any]

# In-memory storage for demo purposes
report_db = {}

@app.post("/report/update")
async def update_report(payload: ReportPayload):
    report_db[payload.module_name] = payload.data_payload
    return {"status": "success", "module": payload.module_name}

@app.get("/report/{module_name}")
async def get_report(module_name: str):
    return report_db.get(module_name, {"error": "Not found"})

## 2. FRONTEND API CONSUMPTION (api_service.js)
# How the Frontend talks to the Backend.

async function fetchReport(moduleName) {
    const response = await fetch(`/report/${moduleName}`);
    const data = await response.json();
    
    // Pass to the ReportGenerator.js from our project
    renderDashboard(data);
}

function renderDashboard(data) {
    // Integration point for ReportGenerator.js
    console.log("Rendering report for:", data);
}
